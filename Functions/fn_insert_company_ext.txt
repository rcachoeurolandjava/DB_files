-- Author: Reygie
-- Purpose: insert data to company_ext table and returns the number of row/s affected
-- Modified by:
-- Modification Date:

CREATE OR REPLACE FUNCTION selenium_data.fn_insert_company_ext(
	p_company_ext text DEFAULT NULL::text)
    RETURNS integer
    LANGUAGE 'plpgsql'

    COST 100
    VOLATILE 
AS $BODY$

DECLARE ret integer;
BEGIN
    INSERT INTO selenium_data.company_ext (ext) VALUES (p_company_ext);
	GET DIAGNOSTICS ret = ROW_COUNT;
	RETURN ret;
END;

$BODY$;
