-- Author: Reygie
-- Purpose: Insert data to crawler_config table and returns number of rows affected
-- Modified by:
-- Modification Date:

CREATE OR REPLACE FUNCTION selenium_data.fn_insert_crawler_config(
	p_id_value bigint DEFAULT NULL::bigint,
	p_config json DEFAULT NULL::json)
    RETURNS integer
    LANGUAGE 'plpgsql'

    COST 100
    VOLATILE 
AS $BODY$
DECLARE ret integer;
BEGIN
    INSERT INTO selenium_data.crawler_config (id, config) VALUES (p_id_value, p_config);
	GET DIAGNOSTICS ret = ROW_COUNT;
	RETURN ret;
END;
$BODY$;
