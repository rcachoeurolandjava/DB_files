CREATE OR REPLACE FUNCTION selenium_data.fn_update_euroland_code(
	p_euroland_code_value text DEFAULT NULL::text,
	p_crawled_name_value text DEFAULT NULL::text)
    RETURNS integer
    LANGUAGE 'plpgsql'

    COST 100
    VOLATILE 
AS $BODY$

DECLARE ret integer;
BEGIN
    UPDATE selenium_data.detailed_crawled_data
	SET company_code = p_euroland_code_value 
	WHERE source_name = p_crawled_name_value;
	GET DIAGNOSTICS ret = ROW_COUNT;
	RETURN ret;
END;

$BODY$;
