-- Author: Reygie
-- Purpose: insert data to crawl_data and returns number of rows affected
-- Modified by:
-- Modification Date:

CREATE OR REPLACE FUNCTION selenium_data.fn_insert_crawled_data(
	p_source_id_value bigint DEFAULT NULL::bigint,
	p_json_data_value json DEFAULT NULL::json)
    RETURNS integer
    LANGUAGE 'plpgsql'

    COST 100
    VOLATILE 
AS $BODY$
DECLARE ret integer;
BEGIN
    INSERT INTO selenium_data.crawl_data (source_id, json_data) VALUES (p_source_id_value, p_json_data_value);
	GET DIAGNOSTICS ret = ROW_COUNT;
	RETURN ret;
END;
$BODY$;
